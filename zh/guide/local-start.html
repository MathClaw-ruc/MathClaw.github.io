本地安装 ​

安装依赖 ​

bash
python -m venv .venv
. .venv/bin/activate
pip install -e .

准备工作区 ​

bash
mkdir -p workspace/sessions
mkdir -p workspace/memory
mkdir -p workspace/cron

启动 gateway ​

bash
mathclaw gateway \
  --workspace ./workspace \
  --channel wecom

启动控制台 ​

bash
cd console
MATHCLAW_CONSOLE_WORKSPACE=../workspace python serve.py
